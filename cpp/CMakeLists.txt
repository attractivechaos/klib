buildAndPackageLib("${PROJECT_NAME}-cpp"
	COMPONENT "cpp_headers"
	DESCRIPTION "${PROJECT_DESCRIPTION}\n\nHeader-only libraries for C++"
	CMAKE_EXPORT_NAMESPACE "${PROJECT_NAME}"
	PUBLIC_INCLUDES "${CMAKE_CURRENT_LIST_DIR}"
	INSTALL_INCLUDE_DIR "${CMAKE_INSTALL_INCLUDEDIR}/${PROJECT_NAME}"
)

list(APPEND CPACK_COMPONENT_CPP_HEADERS_DEPENDS "headers")
set(CPACK_COMPONENT_CPP_HEADERS_DEPENDS "${CPACK_COMPONENT_CPP_HEADERS_DEPENDS}" PARENT_SCOPE)
